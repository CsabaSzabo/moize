Example
import moize from 'moize';

// standard implementation
const fn = (foo, bar) => {
 return `${foo} ${bar}`;
};
const memoizedFn = moize(fn);

// implementation with options
const fn = async (id) => {
 return get(`http://foo.com/${id}`);
};
const memoizedFn = moize(fn, {
 isPromise: true,
 maxSize: 5
});
